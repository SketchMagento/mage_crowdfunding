
================================================================

Create the xml file to make the module visible to magento
-- app/etc/modules/SMage_InstAttrib.xml
<?xml version="1.0"?>
<config>
    <modules>
        <SMage_InstAttrib>
            <active>true</active>
            <codePool>local</codePool>
        </SMage_InstAttrib>
    </modules>
</config>

===============================================================

After that create the config.xml file in the SMage/InstAttrib/etc folder.
<?xml version="1.0"?>
<config>
        <modules>
            <SMage_InstAttrib>
                <version>0.0.1</version>
            </SMage_InstAttrib>
        </modules>
        <global>       
            <resources>
                <add_product_attribute>
                    <setup>
                        <module>SMage_InstAttrib</module>
                        <class>Mage_Catalog_Model_Resource_Setup</class>
                    </setup>
                    <connection>
                        <use>core_setup</use>
                    </connection>
                </add_product_attribute>
            </resources>
        </global>
</config>
